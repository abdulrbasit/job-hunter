%% Override cvevent so date sits left and location sits right-most within the column.
%% Original AltaCV puts both into 0.5\linewidth makeboxes, which centers location.
\renewcommand{\cvevent}[4]{%
  {\large\color{emphasis}#1\par}%
  \smallskip\normalsize
  \ifstrequal{#2}{}{}{\textbf{\color{accent}#2}\par\smallskip}%
  \ifstrequal{#3}{}{}{%
    {\small%
      \BeginAccSupp{method=pdfstringdef,ActualText={\datename:}}\cvDateMarker\EndAccSupp{}~#3%
      \ifstrequal{#4}{}{}{%
        \hfill%
        \BeginAccSupp{method=pdfstringdef,ActualText={\locationname:}}\cvLocationMarker\EndAccSupp{}~#4%
      }%
      \par%
    }%
  }%
  \medskip%
}

  \tikz[baseline]\node[
    anchor=base,
    draw=accent,
    rounded corners,
    inner xsep=0.9ex,
    inner ysep=0.75ex,
    text height=1.5ex,
    text depth=.25ex
  ] {\small #1};\hspace{2pt}%
}
